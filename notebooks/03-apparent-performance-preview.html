retrieve_estimates <- function(model, combo_vec) {
  model |>
    tidy() |>
    select(term, estimate) |>
    rename_with(
      ~ paste0(
        "", combo_vec,
        recycle0 = TRUE
      ),
      .cols = estimate
    ) |>
    mutate(term = str_remove(term, "TRUE"))
